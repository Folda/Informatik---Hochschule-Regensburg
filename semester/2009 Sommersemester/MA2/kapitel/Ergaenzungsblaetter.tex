\subsection{Aufgabe E5}
\label{sec:Ergaenzungsblaetter_E5L}
Lösung zu Aufgabe \vref{sec:Ergaenzungsblaetter_E5}.

\begin{itemize}
\item 6 Möglichkeiten die ersten beiden Ziffern mit Summe 7 zu wählen
\item 10-stellige Zahl mit genau 2 9en und Ziffern 1-9
		\[\binom{10}{2} \mal 8^8\]
		$\binom{10}{2}$ ist die Anzahl der Möglichkeiten Plätze für 9en auszusuchen\\
		$8$ ist die Anzahl der Möglichkeiten ohne 9\\
		$\space^8$ ist die Anzahl der Stellen ohne 9

\item 10-stellige Zahl mit genau einer 9
		\[\binom{10}{1} \mal 8^9\]

\item 10-stellige Zahl ohne 9en
		\[\binom{10}{0} \mal 8^{10}\]
\end{itemize}
da niemals 2 neunen und 1 neun auftreten muss man addieren
\[\Ra 8^8 \mal 6 \mal \binom{10}{2} + 8^9 \mal 6 \mal \binom{10}{1} \mal 8^{10} \mal 6 \mal \underbrace{\binom{10}{0}}_{= 1}\]
\[\Ra 6 \mal \left(\binom{10}{2} \mal 8^9 + \binom{10}{1} \mal 8^9 + \binom{10}{0} \mal 8^{10}\right)\]
\[= 6 \mal 8^8 \underbrace{45 + 80 + 64}_{189}\]
